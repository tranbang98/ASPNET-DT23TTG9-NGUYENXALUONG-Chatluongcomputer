# Di chuyển đến thư mục gốc repo (nếu chưa ở đó)
Set-Location "D:\ASPNET-DT23TTG9-NGUYENXALUONG-Chatluongcomputer"

# Tìm tất cả thư mục (bao gồm thư mục con)
$allDirs = Get-ChildItem -Recurse -Directory

foreach ($dir in $allDirs) {
    # Nếu thư mục không có file nào bên trong
    if (-not (Get-ChildItem -Path $dir.FullName -Recurse)) {
        # Tạo file .gitkeep
        New-Item -Path "$($dir.FullName)\.gitkeep" -ItemType File -Force | Out-Null
    }
}
